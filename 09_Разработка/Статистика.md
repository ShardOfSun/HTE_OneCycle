# –ú–æ—Ç–∏–≤–∞—Ü–∏–∏
```dataviewjs

let targets = {
  "#1A": 1,
  "#1C": 1,
  "#2A": 1,
  "#2C": 1,
  "#3A": 1  
};

// === –®–∞–≥ 2: –°—á–∏—Ç–∞–µ–º —Å–∫–æ–ª—å–∫–æ —Ç–µ–≥–æ–≤ —Ä–µ–∞–ª—å–Ω–æ –µ—Å—Ç—å ===
let tagCounts = {};
for (let page of dv.pages()) {
  if (!page.file.tags) continue;
  for (let tag of page.file.tags) {
    if (!tagCounts[tag]) tagCounts[tag] = 0;
    tagCounts[tag]++;
  }
}

// === –®–∞–≥ 3: –§–æ—Ä–º–∏—Ä—É–µ–º —Ç–∞–±–ª–∏—Ü—É ===

  let actual = tagCounts[tag] || 0;
  let percent = ((actual / target) * 100).toFixed(1);
  let status = actual >= target ? "‚úÖ" : "üü°";
  rows.push([tag, target, actual, `${percent}%`, status]);
}

// === –®–∞–≥ 4: –í—ã–≤–æ–¥–∏–º —Ç–∞–±–ª–∏—Ü—É ===
dv.table(["–¢–µ–≥", "–ù—É–∂–Ω–æ", "–ï—Å—Ç—å", "–ì–æ—Ç–æ–≤–æ (%)", "–°—Ç–∞—Ç—É—Å"], rows);
```
